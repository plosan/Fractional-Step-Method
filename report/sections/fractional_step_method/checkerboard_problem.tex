\begin{equation} \label{eq:velocity_checkerboard_problem}

            u^{n+1} &= u^p - \frac{\Delta t}{\rho} p_x^{n+1} \\
            v^{n+1} &= v^p - \frac{\Delta t}{\rho} p_y^{n+1} \\
            w^{n+1} &= w^p - \frac{\Delta t}{\rho} p_z^{n+1} \\
        \end{aligned}
    \right.
\end{equation}

\end{equation}
